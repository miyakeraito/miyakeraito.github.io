# しりとりゲーム # 単語リスト words = ["りんご", "ごりん", "んぱん", "んたん", "んまち", "ちかん", "んん"] def main(): print("しりとりゲームを始めます。単語の最後の文字を使って次の単語を考えてください。") current_word = None while True: if current_word is None: # 最初の単語を選ぶ current_word = input("最初の単語: ") else: # 次の単語をプレイヤーに入力させる next_letter = current_word[-1] player_word = input(f"{next_letter}から始まる単語: ") if player_word.lower() == "exit": print("ゲームを終了します。") break if player_word in words and player_word[0] == next_letter: current_word = player_word else: print("正しい単語を入力してください。") if __name__ == "__main__": main()
